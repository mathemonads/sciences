% ket linearity
\begin{answer}[Inner product]
	Consider any ket $\ket{C}$ and bra $\bra{A}$.
	\begin{enumerate}
		\item To show $\left\{\bra{A} + \bra{B}\right\} \ket{C} = \braket{A \mid C} + \braket{B \mid C}$, let $\ket{A}$ and $\ket{B}$ be ket-vectors whose bra-vectors are $\bra{A}$ and $\bra{B}$ respectively.

		      There is a ket-vector $\ket{D} = \ket{A} + \ket{B}$.
		      Its bra vector $\bra{D}$ is given by $\bra{D} = \bra{A} + \bra{B}$.
		      Interchanging bras and kets yields the following.
		      \begin{align}\label{eqn:interchange}
			      \braket{D \mid C} = \braket{C \mid D}^* = \left(\braket{C \mid A} + \braket{C \mid B}\right)^* = \braket{C \mid A}^* + \braket{C \mid B}^*
		      \end{align}
		      The final equation follows from linearity of complex conjugation.
		      Complex conjugation of each term in the right-hand expression in Equation \eqref{eqn:interchange} yields $\braket{D \mid C} = \braket{A \mid C} + \braket{B \mid C}$.
		\item To show that the complex number $\braket{A \mid A}$ is a real number for any ket-vector $\ket{A}$, recall that interchanging bras and kets yields a complex number $\braket{A \mid A} = \braket{A \mid A}^*$ that is its own complex conjugate.
		      It follows that its imaginary component is zero and thus that it is real.
	\end{enumerate}
\end{answer}

% calculation
\begin{answer}[Component-wise inner product]
	To show the operation given for any natural number $n$ in terms of components by
	\begin{align*}
		\braket{B \mid A} & =
		\begin{pmatrix}
			\beta_1^* & \beta_2^* & \cdots & \beta_{n-1}^* & \beta_n^*
		\end{pmatrix}
		\begin{pmatrix}
			\alpha_1     \\
			\alpha_2     \\
			\cdots       \\
			\alpha_{n-1} \\
			\alpha_n
		\end{pmatrix}                                                                                                           \\
		                  & = \beta_1^* \alpha_1 + \beta_2^* \alpha_2 + \cdots + \beta_{n-1}^* \alpha_{n-1} + \beta_n^* \alpha_n
	\end{align*}
	is an inner product, observe that it is sesquilinear
	\begin{align*}
		\bra{C} \{ \ket{A} + \ket{B} \} & = \sum_{k=1}^n \gamma_k^* \left(\alpha_k+\beta_k\right) = \sum_{i=1}^n \gamma_i^* \alpha_i + \sum_{j=1}^n \gamma_j^* \beta_j =
		\braket{C \mid A} + \braket{C \mid B}.
	\end{align*}
	and that interchanging bras and kets yields
	\begin{align}
		\braket{B \mid A} = \sum_{k=1}^n \beta_k^* \alpha_k = \left(\sum_{k=1}^n \alpha_k^* \beta_k\right)^* = \braket{A \mid B}^*
	\end{align}
	since $\left(\alpha_k^* \beta_k\right)^* = \beta_k^* \alpha_k$ holds for all $k \leq n$.

	Observe $\dsp \bra{B} \{ z \ket{A} \} = \sum_{k=1}^n \beta_k^* \left(z \alpha_k\right) = z \sum_{k=1}^n \beta_k^* \alpha_k = z \braket{B \mid A}$.
	Conjugate-linearity in the first argument follows immediately from the definition.
	Any ket-vector $\ket{A}$ satisfies $\dsp \braket{A \mid A} = \sum_{k=1}^n \alpha_k^* \alpha_k = \sum_{k=1}^n \left|\alpha_k\right|^2 \geq 0$, and the equation $\braket{A \mid A} = 0$ holds iff $\ket{A} = \ket{0}$.
\end{answer}

% orthogonality of u and d
\begin{answer}[Orthogonality of $r$ and $l$]
	The inner product of the vectors $\dsp \ket{r} = \frac{1}{\sqrt{2}} \ket{u} + \frac{1}{\sqrt{2}} \ket{d}$ and $\dsp \ket{l} = \frac{1}{\sqrt{2}} \ket{u} - \frac{1}{\sqrt{2}} \ket{d}$ is $\frac{1}{2} - \frac{1}{2} = 0$, since $\ket{u}$ and $\ket{d}$ are mutually orthonormal.
	Therefore, the vector $\ket{r}$ is orthogonal to $\ket{l}$.
\end{answer}

% components of bra are conjugated
\begin{answer}[Orthonormality of $i$ and $o$]
	Let $\ket{r} = \frac{1}{\sqrt{2}} \ket{u} + \frac{1}{\sqrt{2}} \ket{d}$, $\ket{l} = \frac{1}{\sqrt{2}} \ket{u} - \frac{1}{\sqrt{2}} \ket{d}$, $\ket{i} = \frac{1}{\sqrt{2}} \ket{u} + \frac{i}{\sqrt{2}} \ket{d}$ and $\ket{o} = \frac{1}{\sqrt{2}} \ket{u} - \frac{i}{\sqrt{2}} \ket{d}$.
	Since $\ket{u}$ and $\ket{d}$ are mutually orthonormal, we obtain orthogonality as shown below.
	\begin{align*}
		\braket{i \mid o} & = \frac{1}{2} + \left(\frac{i}{\sqrt{2}}\right)^* \left(-\frac{i}{\sqrt{2}}\right) = 0
	\end{align*}
	By orthonormality, we also obtain the following equations.
	\begin{align*}
		\braket{o \mid u} \braket{u \mid o} & = \frac{1}{\sqrt{2}} \frac{1}{\sqrt{2}} = \frac{1}{2}                                     \\
		\braket{o \mid d} \braket{d \mid o} & = \left(-\frac{i}{\sqrt{2}}\right)^* \cdot \left(-\frac{i}{\sqrt{2}}\right) = \frac{1}{2} \\
		\braket{i \mid u} \braket{u \mid i} & = \frac{1}{\sqrt{2}} \frac{1}{\sqrt{2}} = \frac{1}{2}                                     \\
		\braket{i \mid d} \braket{d \mid i} & = \left(\frac{i}{\sqrt{2}}\right)^* \left(\frac{i}{\sqrt{2}}\right) = \frac{1}{2}
	\end{align*}
	Similarly, we obtain the following equations.
	\begin{align*}
		\braket{o \mid r} \braket{r \mid o} & = \left(\frac{1}{2} + (-\frac{i}{\sqrt{2}})^* \frac{1}{\sqrt{2}}\right)\left(\frac{1}{2} + \frac{1}{\sqrt{2}}(-\frac{i}{\sqrt{2}}) \right) = \frac{1}{4} + \frac{-i^2}{4} = \frac{1}{2} \\
		\braket{o \mid l} \braket{l \mid o} & = \left(\frac{1 + (-i)^*(-1)}{2}\right)\left(\frac{1 + (-1)(-i)}{2}\right) = \frac{1-i^2}{4} = \frac{1}{2}                                                                              \\
		\braket{i \mid r} \braket{r \mid i} & = \frac{\left(1 -i\right)\left(1 + i\right)}{4} = \frac{1 - i^2}{4} = \frac{1}{2}                                                                                                       \\
		\braket{i \mid l} \braket{l \mid i} & = \frac{\left(1 + i\right)\left(1 - i\right)}{4} = \frac{1 + 1}{4} = \frac{1}{2}
	\end{align*}
	Among pairs of orthogonal vectors satisfying all of the eight equations shown above, the pair of orthogonal vectors $\ket{i}$ and $\ket{o}$ is not strictly unique.
\end{answer}

% calculation
\begin{answer}[Canonicity of $i$ and $o$]
	Let $\ket{u}$ and $\ket{d}$ be an orthonormal basis of a single spin-$\frac{1}{2}$ system.
	Let $\ket{i} = \alpha \ket{u} + \beta \ket{d}$ and $\ket{o} = \gamma \ket{u} + \delta \ket{d}$ be given in terms of unknown components $\alpha$, $\beta$, $\gamma$, and $\delta$.
	Suppose they are a pair of orthonormal vectors satisfying the equations of the previous exercise.
	Note $\braket{u \mid i} = \alpha$, $\braket{d \mid i} = \beta$, $\braket{u \mid o} = \gamma$, and $\braket{d \mid o} = \delta$.
	Their complex conjugates are $\braket{i \mid u} = \alpha^*$, $\braket{i \mid d} = \beta^*$, $\braket{o \mid u} = \gamma^*$, and $\braket{o \mid d} = \delta^*$.
	\begin{enumerate}
		\item Calculation yields $\alpha^* \alpha = \braket{i \mid u} \braket{u \mid i} = \frac{1}{2}$ and $\beta^* \beta = \braket{i \mid d} \braket{d \mid i} = \frac{1}{2}$.
		      Similarly, it follows that $\gamma^* \gamma = \braket{o \mid u} \braket{u \mid o} = \frac{1}{2}$ and $\delta^* \delta = \braket{o \mid d} \braket{d \mid o} = \frac{1}{2}$.
		      Thus, there is a set $\left\{\theta_j \in \R \mid 1 \leq j \leq 4 \right\}$ of phases such that $\alpha = \frac{1}{\sqrt{2}} e^{i \theta_1}$, $\beta = \frac{1}{\sqrt{2}} e^{i \theta_2}$, $\gamma = \frac{1}{\sqrt{2}} e^{i \theta_3}$, and $\delta = \frac{1}{\sqrt{2}} e^{i \theta_4}$.
		\item Recall $\ket{r} = \frac{1}{\sqrt{2}}\left(\ket{u} + \ket{d}\right)$.

		      Since $\braket{r \mid i} = \frac{1}{\sqrt{2}} \left(\alpha + \beta\right)$ and $\braket{ i \mid r} = \frac{1}{\sqrt{2}}\left(\alpha^* + \beta^*\right)$, it follows that $\frac{1}{2} = \braket{i \mid r} \braket{r \mid i} = \frac{1}{2}\left(\alpha^* \alpha + \beta^* \beta + \alpha^* \beta + \alpha \beta^*\right)$ and thus that $\alpha^* \beta + \alpha \beta^* = 0$.
		      Since $\braket{o \mid r} \braket{r \mid o} = \frac{1}{2}\left(\gamma^* \gamma + \delta^* \delta + \gamma^* \delta + \gamma \delta^*\right)$, it follows that $\gamma^* \delta + \gamma \delta^* = 0$.
		\item Since $\left(\alpha^* \beta\right)^* = \alpha \beta^*$, the above result implies that $\alpha^* \beta$ is a pure imaginary number under the convention that $0$ is also pure imaginary.
		      Similarly, the number $\gamma^* \delta$ is pure imaginary under the same convention.
	\end{enumerate}
	Since $\alpha^* \beta$ is pure imaginary, the coefficients $\alpha$ and $\beta$ cannot both be real.
	Similarly, the coefficients $\gamma$ and $\delta$ cannot both be real.
\end{answer}

% characteristic polynomial
\begin{answer}[Fundamental theorem of quantum mechanics]
	To show for any Hermitian operator $L$ on an $N$-dimensional complex vector space that an orthonormal basis of $N$ vectors can be constructed from the eigenvectors of $L$, consider the base case in which $N=0$.
	Since $H = \left\{0\right\}$, the empty set is an orthonormal basis of $N$ vectors and the result follows.
	For the sake of induction, suppose for any Hermitian operator on any $(N-1)$-dimensional complex vector space that an orthonormal basis of $(N-1)$ vectors can be constructed from its eigenvectors.

	Let $L$ be a Hermitian operator on an $N$-dimensional complex vector space $H$.
	Since $H$ is finite-dimensional over $\C$, the characteristic polynomial $p$ of $L$ has degree $N$.
	By the fundamental theorem of algebra, the characteristic polynomial of $L$ has a root and hence $L$ has at at least one eigenvalue $\lambda_n \in \C$.
	Since $L$ is Hermitian, the eigenvalue $\lambda_n$ is real.
	Let $\ket{\lambda_n} \in \p{H \setminus \{0\}}$ be an eigenvector of $\lambda_n$, $V_n = \op{span}\left\{\ket{\lambda_n}\right\}$, and $H_{n-1} = V_n^\perp$ be the orthogonal complement of $V_n$ in $H$.
	Since $L \ket{\lambda_n} = \lambda_n \ket{\lambda_n}$, interchanging bras and kets yields $\bra{\lambda_n} L = \lambda_n \bra{\lambda_n}$ since $L$ is Hermitian and $\lambda_n$ is real.
	If $\ket{A} \in H_{n-1}$ and hence $\braket{\lambda_n \mid A} = 0$, then it satisfies $\braket{\lambda_n \mid L A} = \lambda_n \braket{\lambda_n \mid A} = 0$.
	Hence, $L\p{H_{n-1}} \subseteq H_{n-1}$.
	Thus, the linear operator $L$ is defined on the $(N-1)$-dimensional complex vector space $H_{n-1}$ and it is Hermitian on $H_{n-1}$.
	By the inductive hypothesis, an orthonormal basis $B_{n-1}$ can be constructed from the eigenvectors of $L$ on $H_{n-1}$.
	Thus, the union $\dsp B_{n-1} \cup \left\{\frac{1}{\|\ket{\lambda_n}\|} \ket{\lambda_n} \right\}$ is an orthonormal basis of $H$ consisting of eigenvectors of $L$.
	Therefore, the result follows by induction.
\end{answer}

\begin{answer}[Uniqueness of Pauli matrix]
	Let $\sigma_z$ be a linear operator satisfying the equations below.
	\begin{align}\label{eqn:sz}
		\sigma_z
		\begin{pmatrix}
			1 \\
			0
		\end{pmatrix}
		=
		\begin{pmatrix} 1 \\ 0 \end{pmatrix}
		\qquad
		\sigma_z
		\begin{pmatrix}
			0 \\
			1
		\end{pmatrix}
		= -
		\begin{pmatrix}
			0 \\
			1
		\end{pmatrix}
	\end{align}
	It follows from the first equation above that $\p{\sigma_z}_{11} = 1$ and $\p{\sigma_z}_{21} = 0$.
	It follows from the second equation that $\p{\sigma_z}_{12} = 0$ and $\p{\sigma_z}_{22} = -1$.
	By construction of $\sigma_z = \begin{pmatrix} 1 & 0 \\ 0 & -1 \end{pmatrix}$, it satisfies Equations \eqref{eqn:sz}.
	It is the unique matrix satisfying them.
\end{answer}

Denote the standard Pauli matrices as follows.
\begin{align}
	\sigma_x =
	\begin{pmatrix}
		0 & 1 \\
		1 & 0
	\end{pmatrix}
	\qquad \sigma_y =
	\begin{pmatrix}
		0 & -i \\
		i & 0
	\end{pmatrix}
	\qquad \sigma_z =
	\begin{pmatrix}
		1 & 0  \\
		0 & -1
	\end{pmatrix}
\end{align}

Their transposes are given below.
\begin{align*}
	\trans{\sigma_x} =
	\begin{pmatrix}
		0 & 1 \\
		1 & 0
	\end{pmatrix}
	= \sigma_x
	\qquad
	\trans{\sigma_y} =
	\begin{pmatrix}
		0  & i \\
		-i & 0
	\end{pmatrix}
	= - \sigma_y
	\qquad
	\trans{\sigma_z} =
	\begin{pmatrix}
		1 & 0  \\
		0 & -1
	\end{pmatrix}
	= \sigma_z
\end{align*}
The standard Pauli matrices $\sigma_x$ and $\sigma_z$ are hence symmetric, while $\sigma_y$ is antisymmetric.

Their conjugate transposes are given below.
\begin{align*}
	\sigma_x^\dagger =
	\begin{pmatrix}
		0 & 1 \\
		1 & 0
	\end{pmatrix}
	= \sigma_x
	\qquad
	\sigma_y^\dagger =
	\begin{pmatrix}
		0 & -i \\
		i & 0
	\end{pmatrix}
	= \sigma_y
	\qquad
	\sigma_z^\dagger =
	\begin{pmatrix}
		1 & 0  \\
		0 & -1
	\end{pmatrix}
	= \sigma_z
\end{align*}
Hence, each is Hermitian.

\begin{remark}[Multiplication of the standard Pauli matrices]\label{rem:Pauli}
	The standard Pauli matrices satisfy the relations below.
	\begin{align*}
		\sigma_z \sigma_z =
		\begin{pmatrix}
			1 & 0 \\
			0 & 1
		\end{pmatrix}
		 & = I \qquad
		\sigma_x \sigma_y =
		\begin{pmatrix}
			i & 0  \\
			0 & -i
		\end{pmatrix}
		= i \sigma_z
		\quad
		\sigma_y \sigma_x =
		\begin{pmatrix}
			-i & \phantom{+} 0 \\
			0  & \phantom{+} i
		\end{pmatrix}
		= -i \sigma_z  \\
		\sigma_x \sigma_x =
		\begin{pmatrix}
			1 & 0 \\
			0 & 1
		\end{pmatrix}
		 & = I \qquad
		\sigma_y \sigma_z =
		\begin{pmatrix}
			0 & \phantom{+} i \\
			i & \phantom{+} 0
		\end{pmatrix}
		= i \sigma_x
		\quad
		\sigma_z \sigma_y =
		\begin{pmatrix}
			0  & -i \\
			-i & 0
		\end{pmatrix}
		= - i \sigma_x \\
		\sigma_y \sigma_y =
		\begin{pmatrix}
			1 & 0 \\
			0 & 1
		\end{pmatrix}
		 & = I \qquad
		\sigma_z \sigma_x =
		\begin{pmatrix}
			0  & 1 \\
			-1 & 0
		\end{pmatrix}
		= i \sigma_y \quad
		\sigma_x \sigma_z =
		\begin{pmatrix}
			0 \phantom{+} & -1 \\
			1 \phantom{+} & 0
		\end{pmatrix}
		= - i \sigma_y
	\end{align*}
	For any $i$ and $j$ in $\left\{x,y,z\right\}$, the standard Pauli matrices thus satisfy the equation below.
	\begin{align*}
		\sigma_i \sigma_j & = \delta_{ij} I + i \sum_k \varepsilon_{ijk} \sigma_k
	\end{align*}
\end{remark}

The above remark yields $[\sigma_x, \sigma_x] = [\sigma_y, \sigma_y] = [\sigma_z, \sigma_z] = 0$.

Denote the vector of Pauli matrices by $\sigma = \begin{pmatrix} \sigma_x, & \sigma_y, & \sigma_z \end{pmatrix}$.

For any vector $\vec{n} \in \R^3$, denote the \emph{(standard) Pauli operator in the direction of $\vec{n}$}
\begin{equation}\label{eqn:Pauli}
	n_x \sigma_x + n_y \sigma_y + n_z \sigma_z
	=
	\begin{pmatrix}
		n_z             & \p{n_x - i n_y} \\
		\p{n_x + i n_y} & - n_z
	\end{pmatrix}
\end{equation}
by $\sigma_{\vec{n}}$ or, equivalently, by $\vec{n} \cdot \sigma$.
The operator $\sigma_{\vec{n}}$ is Hermitian since the Pauli matrices are Hermitian and the components of $\vec{n}$ are real.

\begin{proposition}[Product of Pauli operators]\label{prop:Pauli}
	For any vectors $\vec{m}$ and $\vec{n}$ in $\R^3$, the following equation holds.
	\begin{align*}
		\left(\vec{m} \cdot \sigma\right) (\vec{n} \cdot \sigma) & =
		\left(\vec{m} \cdot \vec{n}\right) I + i \left(\vec{m} \times \vec{n}\right) \cdot \sigma
	\end{align*}
\end{proposition}
\begin{proof}
	By Remark \ref{rem:Pauli}, $\dsp S = (\vec{m} \cdot \sigma) (\vec{n} \cdot \sigma) = \sum_{i,j} m_i n_j \sigma_i \sigma_j$ is calculated below.
	\begin{align*}
		S & = \sum_{i, j} m_i n_j \p{\delta_{ij} I + i \sum_k \varepsilon_{ijk} \sigma_k} = \sum_{i, j} \delta_{ij} m_i n_j I + i \sum_{k} \p{\sum_{i,j} \varepsilon_{ijk} m_i n_j} \sigma_k
	\end{align*}
	This yields $\dsp S = \sum_{k} (m_k n_k) I + i \sum_{k} (\vec{m} \times \vec{n})_k \sigma_k = \left(\sum_{k} m_k n_k\right) I + i \sum_{k} (\vec{m} \times \vec{n})_k \sigma_k$ and therefore $\dsp S  = \left(\vec{m} \cdot \vec{n}\right) I + i \left(\vec{m} \times \vec{n}\right) \cdot \sigma$.
\end{proof}

For any unit vector $\hat{n} \in S^2$, the operator $\sigma_{\hat{n}} = \hat{n} \cdot \sigma$ is a Hermitian involution since $\left(\hat{n} \cdot \sigma\right)^2 = I$ and it is thus unitary.

% one parameter for each eigenvector 
\begin{answer}[Spin along direction in $x$-$z$ plane after measurement in $z$]
	Consider the linear operator given in some basis as follows.
	\begin{align}
		\sigma_n & =
		\begin{pmatrix}
			\cos \theta & \sin \theta  \\
			\sin \theta & -\cos \theta
		\end{pmatrix}
	\end{align}
	The characteristic polynomial $\det\p{\sigma_n - \lambda I} = \p{\cos \theta - \lambda}\p{- \cos \theta - \lambda} - \sin^2 \theta$ yields eigenvalues $\lambda_1 = 1$ and $\lambda_2 = -1$.
	Assuming the eigenvector $\ket{\lambda_1}$ has the form $\ket{\lambda_1} = \begin{pmatrix} \cos \alpha \\ \sin \alpha \end{pmatrix}$ yields the equation $\sigma_n \ket{\lambda_1} = \lambda_1 \ket{\lambda_1}$, which is expanded below.
	\begin{align}\label{eqn:eigen}
		\begin{pmatrix}
			\cos \theta \cos \alpha + \sin \theta \sin \alpha \\
			\sin \theta \cos \alpha - \cos \theta \sin \alpha
		\end{pmatrix}
		 & =
		\begin{pmatrix}
			\cos \alpha \\
			\sin \alpha
		\end{pmatrix}
	\end{align}
	Since
	$\sigma_n \ket{\lambda_1} =
		\begin{pmatrix}
			\cos\p{\theta - \alpha} \\
			\sin\p{\theta - \alpha}
		\end{pmatrix}$ and the above equations are simultaneously constrained by sine and cosine,
	the angle $\alpha_1$ in the right-hand side must differ from the angle $\theta-\alpha_1$ in the left-hand side by an integer multiple of the common period $2 \pi$.
	Solving for $\alpha_1$ yields $\alpha_1 = \frac{\theta}{2} - \pi k_1$ for integer $k_1 \in \Z$.
	Any integer $k_1$ determines a normalized eigenvector $\ket{\lambda_1} = \begin{pmatrix} \cos \alpha_1 \\ \sin \alpha_1 \end{pmatrix}$, but the choice of integer determines only whether the resulting eigenvector is $\ket{\lambda_1^0} = \begin{pmatrix} \cos \frac{\theta}{2} \\ \sin \frac{\theta}{2} \end{pmatrix}$ or $- \ket{\lambda_1^0}$.

	Setting $\sigma_n \ket{\lambda_2} = - \ket{\lambda_2} = - \begin{pmatrix} \cos \alpha \\ \sin \alpha \end{pmatrix}$ and solving
	$
		\begin{pmatrix}
			\cos\p{\theta - \alpha_2} \\
			\sin\p{\theta - \alpha_2}
		\end{pmatrix}
		=
		\begin{pmatrix}
			- \cos \alpha_2 \\
			- \sin \alpha_2
		\end{pmatrix}
	$
	yields $\alpha_2 = \frac{\theta - \pi}{2} - \pi k_2$ for each integer $k_2 \in \Z$, each of which determines an
	eigenvector $\ket{\lambda_2} = \begin{pmatrix} \cos\p{\alpha_2} \\ \sin\p{\alpha_2} \end{pmatrix}$.
	Each eigenvalue has a $1$-dimensional eigenspace, and any normalized eigenvector is thus unique up to an overall sign.
	It was sensible to assume a single parameter $\alpha$ since the two components of a normalized eigenvector are related by the constraint of unit length.
\end{answer}

% minimal polynomial
\begin{answer}[Spin along arbitrary direction after measurement in $z$]
	Let $n_z = \cos \theta$, $n_x = \sin \theta \cos \phi$, $n_y = \sin \theta \sin \phi$, and $\sigma_n$ be given as follows.
	\begin{align*}
		\begin{pmatrix}
			n_z             & \p{n_x - i n_y} \\
			\p{n_x + i n_y} & - n_z
		\end{pmatrix}
		 & =
		\begin{pmatrix}
			\cos \theta             & e^{-i \phi} \sin \theta \\
			e^{i \phi } \sin \theta & - \cos \theta
		\end{pmatrix}
	\end{align*}
	Since $\sigma_n \sigma_n = I$, the polynomial $\lambda^2 - 1$ annihilates $\sigma_n$ and the minimal polynomial divides $\lambda^2 - 1$.
	Since the eigenvalues of $\sigma_n$ satisfy $\lambda^2 = 1$ and $\Tr \sigma_n = 0$, they are $\lambda_1 = 1$ and $\lambda_2 = -1$.
	Suppose $\ket{\lambda_1} = \begin{pmatrix} \cos \alpha \\ e^{i \beta} \sin \alpha \end{pmatrix}$ solves the equation $\sigma_n \ket{\lambda_1} = \lambda_1 \ket{\lambda_1}$, which is expanded below.
	\begin{align*}
		\begin{pmatrix}
			\cos \theta \cos \alpha + e^{i(\beta - \phi)} \sin \theta \sin \alpha \\
			e^{i \phi} \sin \theta \cos \alpha - e^{i \beta} \cos \theta \sin \alpha
		\end{pmatrix}
		=
		\begin{pmatrix}
			\cos \alpha \\
			e^{i \beta} \sin \alpha
		\end{pmatrix}
	\end{align*}
	This yields $\beta = \phi + 2 \pi k_\beta$ for any integer $k_\beta \in \Z$, $\alpha = \frac{\theta}{2} + \pi k_\alpha$ for any integer $k_\alpha \in \Z$, and choosing $k_\alpha = k_\beta = 0$ yields $\ket{\lambda_1} = \begin{pmatrix} \cos \frac{\theta}{2} \\ e^{i \phi} \sin \frac{\theta}{2} \end{pmatrix}$.
	Similarly, we obtain $\ket{\lambda_2} = \begin{pmatrix} \sin \frac{\theta}{2} \\ - e^{i \phi} \cos \frac{\theta}{2} \end{pmatrix}$.
	Each normalized eigenvector of $\sigma_n$ is unique up to a complex phase.
\end{answer}

\begin{lemma}
	For every rotation $r$ of $\R^3$, there is a unitary operator $R$ on $\C^2$ satisfying
	\begin{align*}
		\forall \hat{m} & \in S^2, & R \left(\hat{m} \cdot \sigma\right) R^\dagger & = \left(r \hat{m}\right) \cdot \sigma
	\end{align*}
	and any unitary operator $R'$ satisfying the above condition is of the form $R' = e^{i \phi} R$.
\end{lemma}
\begin{proof}
	Consider any rotation $r$ of $\R^3$.
	Denote its axis by $\hat{n}$ and angle by $\theta$.

	Let $R = \exp\p{- i \frac{\theta}{2} \hat{n} \cdot \sigma}$.
	It is given by $R = \left(\cos \frac{\theta}{2} \right) I - i \left(\sin \frac{\theta}{2}\right) \hat{n} \cdot \sigma$.
	Let $c = \cos \frac{\theta}{2}$, $s = \sin \frac{\theta}{2}$, $N = \hat{n} \cdot \sigma$, and $M = \hat{m} \cdot \sigma$.
	Calculation yields $R = c I - i s N$ and $R^\dagger = c I + i s N$.
	Substitution yields the equations below.
	\begin{align*}
		R M R^\dagger    & = c^2 M + ics(M N - N M) + s^2 N M N                                                                                                                                                                        \\
		MN               & = \left(\hat{m} \cdot \hat{n}\right) I + i \left(\hat{m} \times \hat{n}\right) \cdot \sigma                                                                                                                 \\
		NM               & = \left(\hat{n} \cdot \hat{m}\right) I -  i \left(\hat{m} \times \hat{n}\right) \cdot \sigma                                                                                                                \\
		MN - NM          & = 2 i \left(\hat{m} \times \hat{n}\right) \cdot \sigma                                                                                                                                                      \\
		\left(NM\right)N & = \left(\hat{n} \cdot \hat{m}\right) \left(\hat{n} \cdot \sigma\right)  - i \Big(\big((\hat{m} \times \hat{n}) \cdot \hat{n}\big) I + i \big((\hat{m} \times \hat{n}) \times \hat{n}\big) \cdot \sigma\Big)
	\end{align*}
	Recall $\left(\hat{m} \times \hat{n}\right) \cdot \hat{n} = 0$ and $\left(\hat{m} \times \hat{n}\right) \times \hat{n} = \left(\hat{m} \cdot \hat{n}\right) \hat{n} - \hat{m}$.
	Substitution yields $\left(NM\right)N = \left(\hat{n} \cdot \hat{m}\right) \left(\hat{n} \cdot \sigma\right) + \left((\hat{n} \cdot \hat{m}) \hat{n} - \hat{m}\right) \cdot \sigma = \left(2(\hat{n} \cdot \hat{m}) \hat{n} - \hat{m}\right) \cdot \sigma$ and also $R M R^\dagger = c^2 M + 2 c s \left(\hat{n} \times \hat{m}\right) \cdot \sigma + s^2 \left(2(\hat{n} \cdot \hat{m}) \hat{n} - \hat{m}\right) \cdot \sigma$.
	Subtraction yields $R M R^\dagger = \left((c^2 - s^2) \hat{m} + 2 c s (\hat{n} \times \hat{m}) + 2 s^2 (\hat{n} \cdot \hat{m}) \hat{n} \right) \cdot \sigma$ and the equation below.
	\begin{align}
		R M R^\dagger & = \left(\hat{m} \cos \theta + (\hat{n} \times \hat{m}) \sin \theta + \hat{n} (\hat{n} \cdot \hat{m}) (1 - \cos \theta) \right) \cdot \sigma
	\end{align}
	Applying Rodrigues's formula yields $R M R^\dagger = \left(r\hat{m} \right) \cdot \sigma$.
	Since $\hat{n} \cdot \sigma$ is Hermitian, the operator $R$ is unitary.

	Suppose $R'$ is also a linear operator that satisfies $R' \left(\hat{m} \cdot \sigma\right) {R'}^{\dagger} = \left(r \hat{m}\right) \cdot \sigma$ for all $\hat{m} \in S^2$.
	Consider any unit vector $\hat{m} \in S^2$, and let $U = R^\dagger R'$.
	Since $R' \left(\hat{m} \cdot \sigma\right) {R'}^\dagger = R \left(\hat{m} \cdot \sigma\right) R^\dagger$, it follows from multiplication that $U \left(\hat{m} \cdot \sigma\right) = \left(\hat{m} \cdot \sigma\right) U$.
	This implies $U \sigma_x = \sigma_x U$, $U \sigma_y = \sigma_y U$, and $U \sigma_z = \sigma_z U$.
	Commutativity of $U$ with $\sigma_z$ implies that $U$ is diagonal, and commutativity of $U$ with $\sigma_x$ then implies $U = a I$ for some $a \in \C$.
	Unitarity of $U$ implies $\left|a\right|=1$ and thus that $a = e^{i \phi}$ for some real $\phi \in \R$.
	The rest follows.
\end{proof}

The above proposition follows from the fact that Pauli matrices transform like ordinary vectors under rotations.

\begin{answer}[Commutator of Hermitian operators]
	Let $M$ and $L$ be Hermitian operators, hence satisfying $M^\dagger = M$ and $L^\dagger = L$.

	Their commutator $[M, L] = M L - L M$ satisfies $i [M, L] = i M L - i L M$ and $[M, L]^\dagger = L^\dagger M^\dagger - M^\dagger L^\dagger = L M - M L = - [M, L]$.
	Thus, the commutator $[M, L]$ is Hermitian if and only if it is zero.

	The Hermitian conjugate $\left(i[M, L]\right)^\dagger = -i \left([M, L]\right)^\dagger$ satisfies $\left(i [M, L]\right)^\dagger = i [M, L]$.
	Therefore, the operator $i [M, L]$ is Hermitian.
\end{answer}

	\begin{enumerate}

	\end{enumerate}
\end{answer}

	Hence, Alice's spin operators send product states to product states.
	Similarly, Bob's spin operators send product states to product states.

	For any product state $\ket{\alpha} \otimes \ket{\beta}$, the expectation value of any component $\sigma_i$ of $\sigma$ is $\braket{\alpha \beta \mid \sigma_i \mid \alpha \beta} =  \braket{\alpha \mid \sigma_i \mid \alpha} \braket{\beta \mid \beta}$; if the state $\ket{\beta}$ is normalized, then $\braket{\alpha \beta \mid \sigma_i \mid \alpha \beta} = \braket{\alpha \mid \sigma_i \mid \alpha}$.
	Similarly, the expectation value of any component of $\tau$ in any product state is exactly the same as it would be in the individual single-spin states.
\end{answer}

\begin{answer}[Uncorrelated measurements on singlet]
	Assume after Charlie has prepared the two spins in the singlet that Bob measures $\tau_y$ and Alice measures $\sigma_x$.

	The action of $\sigma_x \tau_y$ on the singlet state $\ket{s} = \frac{1}{\sqrt{2}} \left(\ket{u d} - \ket{d u}\right)$ is calculated below.
	\begin{align}
		\sigma_x \tau_y \ket{s} = \frac{-i}{\sqrt{2}} \left(\ket{d u} + \ket{u d}\right)
	\end{align}
	The expectation $\braket{s \mid \sigma_x \tau_y \mid s} = \frac{-i}{2} \left(\braket{u d \mid d u} + \braket{u d \mid u d} - \braket{d u \mid d u} - \braket{d u \mid u d}\right)$ is $\braket{s \mid \sigma_x \tau_y \mid s} = 0$.
	Therefore, the outcomes of measurements in the singlet state of $\sigma_x$ and $\tau_y$ are not correlated.
\end{answer}

\begin{answer}
	Assume Charlie prepares the spins in the state $\ket{T_1} = \frac{1}{\sqrt{2}} \left(\ket{u d} + \ket{d u}\right)$.
	The measurement on it of $\sigma_z \tau_z$ is given by $\sigma_z \tau_z \ket{T_1} = \frac{-1}{\sqrt{2}} \left(\ket{u d} + \ket{d u}\right) = - \ket{T_1}$.
	The expectation is calculated below.
	\begin{align*}
		\braket{T_1 \mid \sigma_z \tau_z \mid T_1} & = \frac{-1}{2} \left(\braket{u d \mid u d + d u} + \braket{d u \mid u d + d u }\right) = -1 \\
		\braket{T_1 \mid \sigma_x \tau_x \mid T_1} & = \braket{T_1 \mid T_1} = 1
	\end{align*}
	Thus, the measurements in $y$ are perfectly correlated, while the corresponding measurements in $z$ is perfectly anti-correlated.
